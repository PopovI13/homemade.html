javac ShooterGame.java java ShooterGame import javax.swing.*; import java.awt.*; import java.awt.event.*; import java.util.ArrayList; import java.util.Iterator; import java.util.Random; public class ShooterGame extends JPanel implements ActionListener, KeyListener { // Game constants final int WIDTH = 800; final int HEIGHT = 600; Timer timer; boolean[] keys = new boolean[256]; Player player; ArrayList bullets = new ArrayList<>(); ArrayList enemies = new ArrayList<>(); Random random = new Random(); int score = 0; boolean gameOver = false; public ShooterGame() { setPreferredSize(new Dimension(WIDTH, HEIGHT)); setBackground(Color.BLACK); setFocusable(true); addKeyListener(this); player = new Player(WIDTH / 2 - 20, HEIGHT - 80); timer = new Timer(15, this); // ~60 FPS timer.start(); } public static void main(String[] args) { JFrame frame = new JFrame("Java 2D Shooter"); ShooterGame gamePanel = new ShooterGame(); frame.add(gamePanel); frame.pack(); frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); frame.setVisible(true); } // Main game loop public void actionPerformed(ActionEvent e) { if (!gameOver) { player.update(keys); // Shoot bullets if (keys[KeyEvent.VK_SPACE] && player.canShoot()) { bullets.add(new Bullet(player.x + player.width / 2 - 2, player.y)); player.resetShootTimer(); } player.decreaseShootTimer(); // Update bullets Iterator bulletIter = bullets.iterator(); while (bulletIter.hasNext()) { Bullet b = bulletIter.next(); b.update(); if (b.y < 0) bulletIter.remove(); } // Spawn enemies randomly if (random.nextInt(30) == 0) { enemies.add(new Enemy(random.nextInt(WIDTH - 40), -40)); } // Update enemies Iterator enemyIter = enemies.iterator(); while (enemyIter.hasNext()) { Enemy en = enemyIter.next(); en.update(); if (en.y > HEIGHT) { enemyIter.remove(); continue; } // Check for collision with player if (en.getBounds().intersects(player.getBounds())) { gameOver = true; break; } // Check for collision with bullets Iterator bIter = bullets.iterator(); while (bIter.hasNext()) { Bullet b = bIter.next(); if (b.getBounds().intersects(en.getBounds())) { bIter.remove(); enemyIter.remove(); score += 10; break; } } } } repaint(); } // Drawing everything public void paintComponent(Graphics g) { super.paintComponent(g); // Draw player player.draw(g); // Draw bullets for (Bullet b : bullets) { b.draw(g); } // Draw enemies for (Enemy e : enemies) { e.draw(g); } // Draw score g.setColor(Color.WHITE); g.setFont(new Font("Arial", Font.BOLD, 20)); g.drawString("Score: " + score, 10, 25); // Game over if (gameOver) { g.setFont(new Font("Arial", Font.BOLD, 48)); g.setColor(Color.RED); g.drawString("GAME OVER", WIDTH / 2 - 150, HEIGHT / 2); } } // Key events public void keyPressed(KeyEvent e) { keys[e.getKeyCode()] = true; } public void keyReleased(KeyEvent e) { keys[e.getKeyCode()] = false; } public void keyTyped(KeyEvent e) { } // Player class class Player { int x, y, width = 40, height = 40, speed = 5; int shootCooldown = 0; public Player(int x, int y) { this.x = x; this.y = y; } public void update(boolean[] keys) { if (keys[KeyEvent.VK_LEFT] || keys[KeyEvent.VK_A]) x -= speed; if (keys[KeyEvent.VK_RIGHT] || keys[KeyEvent.VK_D]) x += speed; if (x < 0) x = 0; if (x > WIDTH - width) x = WIDTH - width; } public void draw(Graphics g) { g.setColor(Color.CYAN); g.fillRect(x, y, width, height); } public boolean canShoot() { return shootCooldown <= 0; } public void resetShootTimer() { shootCooldown = 15; // cooldown frames } public void decreaseShootTimer() { if (shootCooldown > 0) shootCooldown--; } public Rectangle getBounds() { return new Rectangle(x, y, width, height); } } // Bullet class class Bullet { int x, y, speed = 10, width = 4, height = 10; public Bullet(int x, int y) { this.x = x; this.y = y; } public void update() { y -= speed; } public void draw(Graphics g) { g.setColor(Color.YELLOW); g.fillRect(x, y, width, height); } public Rectangle getBounds() { return new Rectangle(x, y, width, height); } } // Enemy class class Enemy { int x, y, speed = 2, width = 40, height = 40; public Enemy(int x, int y) { this.x = x; this.y = y; } public void update() { y += speed; } public void draw(Graphics g) { g.setColor(Color.RED); g.fillRect(x, y, width, height); } public Rectangle getBounds() { return new Rectangle(x, y, width, height); } } }
